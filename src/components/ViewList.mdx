## Basic usage

<Playground>
  {
    () => {
      let initList = []
      for (let i = 0; i < 30; i++) {
        initList.push({
          id: Math.random() * 100,
          name: '测试' + Math.random() * 100
        })
      }
      const [list, setList] = React.useState(initList)
      const onLoad = (data) => {
        setList([...list, ...data])
      }
      const refreshList = data => {
        setList([...data])
      }
      const API = (params) => new Promise(resolve => {
        console.log(params)
        setTimeout(() => {resolve(initList)}, 1000)
      })
      return (
          <div style={{height: 600}}>
            <ListView
              getList={onLoad}
              API={API}
              refreshList={refreshList}
            >
             <div>
              {
                list.map((item, index) => (
                 <div key={index}>{item.name + ` ${index}`}</div>
                 ))
              }
              </div>
            </ListView>
          </div>
       )
    }
  }
</Playground>
