## Basic usage

<Playground>
  {
    () => {
      const [visible, setVisible] = React.useState(false)
      return (
      <div>
      <button
        onClick={() => {
          setVisible(!visible)
        }}
      >
        打开
      </button>
      <Picker
        placement='top'
        visible={visible}
        onCancel={() => {
          setVisible(false)
        }}
      >
              {
                Array.from({length: 100}).map((i, v) => (<div key={v}>{v}</div>))
              }
            </Picker>
      </div>)
    }
  }
</Playground>
